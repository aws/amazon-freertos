27 * FreeRTOS_Stream_Buffer.h

29 * A circular character buffer
30 * An implementation of a circular buffer without a length field
31 * If LENGTH defines the size of the buffer, a maximum of (LENGTH-1) bytes can be stored
32 * In order to add or read data from the buffer, memcpy() will be called at most 2 times

69 const size_t uxLower,
70 const size_t uxUpper );
71 static portINLINE size_t uxStreamBufferSpace( const StreamBuffer_t * pxBuffer,
72 const size_t uxLower,
73 const size_t uxUpper )
74 {
75/* Returns the space between uxLower and uxUpper, which equals to the distance minus 1 */
76 size_t uxCount;
77
78 uxCount = pxBuffer->LENGTH + uxUpper - uxLower - 1U;
79
80 if( uxCount >= pxBuffer->LENGTH )
81 {
82 uxCount -= pxBuffer->LENGTH;
83 }
84
85 return uxCount;
86 }

159 static portINLINE void vStreamBufferMoveMid( StreamBuffer_t * pxBuffer,
160 size_t uxCount );
161 static portINLINE void vStreamBufferMoveMid( StreamBuffer_t * pxBuffer,
162 size_t uxCount )
163 {
164/* Increment uxMid, but no further than uxHead */
165 size_t uxSize = uxStreamBufferMidSpace( pxBuffer );
166 size_t uxMoveCount = uxCount;
167
168 if( uxMoveCount > uxSize )
169 {
170 uxMoveCount = uxSize;
171 }
172
173 pxBuffer->uxMid += uxMoveCount;
174
175 if( pxBuffer->uxMid >= pxBuffer->LENGTH )
176 {
177 pxBuffer->uxMid -= pxBuffer->LENGTH;
178 }
179 }

182 static portINLINE BaseType_t xStreamBufferLessThenEqual( const StreamBuffer_t * pxBuffer,
183 const size_t uxLeft,
184 const size_t uxRight );
185 static portINLINE BaseType_t xStreamBufferLessThenEqual( const StreamBuffer_t * pxBuffer,
186 const size_t uxLeft,
187 const size_t uxRight )
188 {
189 BaseType_t xReturn;
190 size_t uxTail = pxBuffer->uxTail;
191
192 /* Returns true if ( uxLeft < uxRight ) */
193 if( ( ( ( uxLeft < uxTail ) ? 1U : 0U ) ^ ( ( uxRight < uxTail ) ? 1U : 0U ) ) != 0U )
194 {
195 if( uxRight < uxTail )
196 {
197 xReturn = pdTRUE;
198 }
199 else
200 {
201 xReturn = pdFALSE;
202 }
203 }
204 else
205 {
206 if( uxLeft <= uxRight )
207 {
208 xReturn = pdTRUE;
209 }
210 else
211 {
212 xReturn = pdFALSE;
213 }
214 }
215
216 return xReturn;
217 }

222 static portINLINE size_t uxStreamBufferGetPtr( StreamBuffer_t * pxBuffer,
223 uint8_t ** ppucData )
224 {
225 size_t uxNextTail = pxBuffer->uxTail;
226 size_t uxSize = uxStreamBufferGetSize( pxBuffer );
227
228 *ppucData = pxBuffer->ucArray + uxNextTail;
229
230 return FreeRTOS_min_uint32( uxSize, pxBuffer->LENGTH - uxNextTail );
231 }
