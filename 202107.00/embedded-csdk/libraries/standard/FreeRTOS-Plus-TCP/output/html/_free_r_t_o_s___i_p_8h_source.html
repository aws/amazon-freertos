85/* Space left at the beginning of a network buffer storage area to store a
86 * pointer back to the network buffer. Should be a multiple of 8 to ensure 8 byte
87 * alignment is maintained on architectures that require it.
88 *
89 * In order to get a 32-bit alignment of network packets, an offset of 2 bytes
90 * would be desirable, as defined by ipconfigPACKET_FILLER_SIZE. So the malloc'd
91 * buffer will have the following contents:
92 * uint32_t pointer; // word-aligned
93 * uchar_8 filler[6];
94 * << ETH-header >> // half-word-aligned
95 * uchar_8 dest[6]; // start of pucEthernetBuffer
96 * uchar_8 dest[6];
97 * uchar16_t type;
98 * << IP-header >> // word-aligned
99 * uint8_t ucVersionHeaderLength;
100 * etc
101 */
102
103 #if ( ipconfigBUFFER_PADDING != 0 )
104 #define ipBUFFER_PADDING ipconfigBUFFER_PADDING
105 #else
106 #define ipBUFFER_PADDING ( 8U + ipconfigPACKET_FILLER_SIZE )
107 #endif
