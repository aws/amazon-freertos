86/*
87 * Look for ulIPAddress in the ARP cache. If the IP address exists, copy the
88 * associated MAC address into pxMACAddress, refresh the ARP cache entry's
89 * age, and return eARPCacheHit. If the IP address does not exist in the ARP
90 * cache return eARPCacheMiss. If the packet cannot be sent for any reason
91 * (maybe DHCP is still in process, or the addressing needs a gateway but there
92 * isn't a gateway defined) then return eCantSendPacket.
93 */
94 eARPLookupResult_t eARPGetCacheEntry( uint32_t * pulIPAddress,
95 MACAddress_t * const pxMACAddress );
