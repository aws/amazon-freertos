222 static portINLINE BaseType_t xSocketValid( Socket_t xSocket )
223 {
224 BaseType_t xReturnValue = pdFALSE;
225
226 /*
227 * There are two values which can indicate an invalid socket:
228 * FREERTOS_INVALID_SOCKET and NULL. In order to compare against
229 * both values, the code cannot be compliant with rule 11.4,
230 * hence the Coverity suppression statement below.
231 */
232 /* coverity[misra_c_2012_rule_11_4_violation] */
233 if( ( xSocket != FREERTOS_INVALID_SOCKET ) && ( xSocket != NULL ) )
234 {
235 xReturnValue = pdTRUE;
236 }
237
238 return xReturnValue;
239 }

393 * // handle the message
394 * return 1;
395 * }
